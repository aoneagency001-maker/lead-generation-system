```python
from typing import Generic, TypeVar, Optional, List

T = TypeVar('T')

class BaseRepository(ABC, Generic[T]):
    """
    –ë–∞–∑–æ–≤—ã–π —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏–π (–ú–ò–ù–ò–ú–ê–õ–¨–ù–´–ô ABC)

    """
    
    @abstractmethod
    async def get(self, id: str) -> Optional[T]:
        pass
    
    @abstractmethod
    async def list(self, skip: int = 0, limit: int = 100) -> List[T]:
        pass
    
    @abstractmethod
    async def create(self, entity: T) -> T:
        pass
    
    @abstractmethod
    async def update(self, id: str, entity: T) -> T:
        pass
    
    @abstractmethod

        pass
```

```python
from typing import Optional, List
from supabase import Client
from backend.core.domain.entities.lead import Lead, LeadStatus
from datetime import datetime

class LeadRepository:
    """
    Concrete Repository –¥–ª—è Lead
    
    –ù–ï –Ω–∞—Å–ª–µ–¥—É–µ—Ç—Å—è –æ—Ç ABC - –ø—Ä—è–º–∞—è —Ä–µ–∞–ª–∏–∑–∞—Ü–∏—è (Google AI –ø–æ–¥—Ö–æ–¥)
    """
    
    def __init__(self, supabase_client: Client):
        self.client = supabase_client
        self.table = "leads"
    
    async def get(self, lead_id: str) -> Optional[Lead]:
        """–ü–æ–ª—É—á–∏—Ç—å –ª–∏–¥ –ø–æ ID"""
        response = self.client.table(self.table).select("*").eq("id", lead_id).execute()
        
        if not response.data:
            return None
        
        return self._to_entity(response.data[0])
    
    async def list(
        self,
        campaign_id: Optional[str] = None,
        status: Optional[LeadStatus] = None,
        skip: int = 0,
        limit: int = 100
    ) -> List[Lead]:
        """–ü–æ–ª—É—á–∏—Ç—å —Å–ø–∏—Å–æ–∫ –ª–∏–¥–æ–≤ —Å —Ñ–∏–ª—å—Ç—Ä–∞–º–∏"""
        query = self.client.table(self.table).select("*")
        
        if campaign_id:
            query = query.eq("campaign_id", campaign_id)
        
        if status:
            query = query.eq("status", status.value)
        
        response = query.range(skip, skip + limit - 1).execute()
        
        return [self._to_entity(row) for row in response.data]
    
    async def create(self, lead: Lead) -> Lead:
        """–°–æ–∑–¥–∞—Ç—å –ª–∏–¥"""
        data = self._to_dict(lead)
        response = self.client.table(self.table).insert(data).execute()
        
        return self._to_entity(response.data[0])
    
    async def update(self, lead_id: str, lead: Lead) -> Lead:
        """–û–±–Ω–æ–≤–∏—Ç—å –ª–∏–¥"""
        data = self._to_dict(lead)
        response = self.client.table(self.table).update(data).eq("id", lead_id).execute()
        
        return self._to_entity(response.data[0])
    
    async def delete(self, lead_id: str) -> bool:
        """–£–¥–∞–ª–∏—Ç—å –ª–∏–¥"""
        response = self.client.table(self.table).delete().eq("id", lead_id).execute()
        return len(response.data) > 0
    
    # === DOMAIN-SPECIFIC QUERIES ===
    
    async def get_qualified_leads(self, campaign_id: str) -> List[Lead]:
        """–ü–æ–ª—É—á–∏—Ç—å –∫–≤–∞–ª–∏—Ñ–∏—Ü–∏—Ä–æ–≤–∞–Ω–Ω—ã–µ –ª–∏–¥—ã"""
        return await self.list(
            campaign_id=campaign_id,
            status=LeadStatus.QUALIFIED
        )
    
    async def get_leads_for_auto_disqualification(self) -> List[Lead]:
```

```python
from typing import List
from backend.core.domain.entities.lead import Lead, LeadStatus
from backend.core.repository.lead_repository import LeadRepository
from backend.core.events.event_bus import EventBus
from backend.core.events.events import LeadQualifiedEvent, LeadDisqualifiedEvent

class LeadQualificationService:
    """
    Service –¥–ª—è –∫–≤–∞–ª–∏—Ñ–∏–∫–∞—Ü–∏–∏ –ª–∏–¥–æ–≤
    
    –û—Ç–≤–µ—á–∞–µ—Ç –∑–∞:
    - –û—Ä–∫–µ—Å—Ç—Ä–∞—Ü–∏—é –∫–≤–∞–ª–∏—Ñ–∏–∫–∞—Ü–∏–∏
    - –ë–∏–∑–Ω–µ—Å-–ø—Ä–∞–≤–∏–ª–∞
    - –ü—É–±–ª–∏–∫–∞—Ü–∏—é —Å–æ–±—ã—Ç–∏–π
    """
    
    def __init__(
        self,
        lead_repo: LeadRepository,
        event_bus: EventBus
    ):
        self.lead_repo = lead_repo
        self.event_bus = event_bus
    
    async def qualify_lead(
        self,
        lead_id: str,
        qualification_score: float
    ) -> Lead:
        """
        –ö–≤–∞–ª–∏—Ñ–∏—Ü–∏—Ä–æ–≤–∞—Ç—å –ª–∏–¥
        
        Business Flow:
        1. –ü–æ–ª—É—á–∏—Ç—å –ª–∏–¥ –∏–∑ —Ä–µ–ø–æ–∑–∏—Ç–æ—Ä–∏—è
        2. –ü—Ä–æ–≤–µ—Ä–∏—Ç—å can_qualify()
        3. –ü—Ä–∏–º–µ–Ω–∏—Ç—å qualify()
        4. –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –≤ –ë–î
        5. –û–ø—É–±–ª–∏–∫–æ–≤–∞—Ç—å —Å–æ–±—ã—Ç–∏–µ
        """
        # 1. Get lead
        lead = await self.lead_repo.get(lead_id)
        if not lead:
            raise ValueError(f"Lead {lead_id} not found")
        
        # 2-3. Business logic (–≤ entity!)
        lead.qualify(qualification_score)
        
        # 4. Save
        lead = await self.lead_repo.update(lead_id, lead)
        
        # 5. Publish event
        await self.event_bus.publish(LeadQualifiedEvent(
            lead_id=lead.id,
            campaign_id=lead.campaign_id,
            qualification_score=qualification_score
        ))
        
        return lead
    
    async def auto_disqualify_stale_leads(self) -> List[Lead]:
        """
        –ê–≤—Ç–æ–º–∞—Ç–∏—á–µ—Å–∫–∞—è –¥–∏—Å–∫–≤–∞–ª–∏—Ñ–∏–∫–∞—Ü–∏—è –∑–∞—Å—Ç—Ä—è–≤—à–∏—Ö –ª–∏–¥–æ–≤
        
        Business Rule: –ï—Å–ª–∏ –ª–∏–¥ > 10 —Å–æ–æ–±—â–µ–Ω–∏–π –∏ score < 0.3
        """
        candidates = await self.lead_repo.get_leads_for_auto_disqualification()
        
        disqualified = []
        for lead in candidates:
            if lead.should_auto_disqualify():  # Business logic –≤ entity!
                lead.disqualify("Auto-disqualified: low engagement")
                await self.lead_repo.update(lead.id, lead)
                
                await self.event_bus.publish(LeadDisqualifiedEvent(
                    lead_id=lead.id,
                    reason="auto_disqualified"
                ))
                
                disqualified.append(lead)
        
        return disqualified
```
